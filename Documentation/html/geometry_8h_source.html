30 inline Eigen::Matrix3d so3_rightJacobian(const Eigen::Vector3d &w) {
31  double w_norm = w.norm();
32  Eigen::Matrix3d w_skew = skewMatrix(w);
33  if (w_norm < 1e-5)
34  return Eigen::Matrix3d::Identity();
35  return Eigen::Matrix3d::Identity() - ((1 - cos(w_norm)) / (w_norm * w_norm)) * w_skew +
36  ((w_norm - sin(w_norm)) / (w_norm * w_norm * w_norm)) * w_skew * w_skew;
37 }

39 inline Eigen::Matrix3d so3_leftJacobian(const Eigen::Vector3d &w) {
40  Eigen::Matrix3d Jl;
41  double w_norm = w.norm();
42  Eigen::Vector3d a = w / w_norm;
43  Jl = sin(w_norm) / w_norm * Eigen::Matrix3d::Identity() + (1 - sin(w_norm) / w_norm) * a * a.transpose() +
44  (1 - cos(w_norm)) / w_norm * skewMatrix(a);
45  return Jl;

111 inline Eigen::Matrix3d directionVector2Rotation(Eigen::Vector3d v,
112  const Eigen::Vector3d &reference_vector = Eigen::Vector3d(1, 0, 0)) {
113 
114  // v = R*reference
115  // Rodrigues's rotation formula gives the result of a rotation of a unit vector a about an axis of
116  // rotation k through the angle θ. We can make use of this by realizing that, in order to bring
117  // a normalized vector a into coincidence with another normalized vector b, we simply need to
118  // rotate a about k=(a+b)/2 by the angle pi.
119  v = v.normalized(); // assure v is normalized
120  Eigen::Vector3d sum = v + reference_vector;
121  Eigen::Matrix3d R = 2.0 * (sum * sum.transpose()) / (sum.transpose() * sum) - Eigen::Matrix3d::Identity();
122  return R;
123 }

131 inline Eigen::Matrix3d exp_so3(const Eigen::Vector3d &v) {
132 
133  double tolerance = 1e-9;
134  double angle = v.norm();
135 
136  Eigen::Matrix3d Rot;
137  if (angle < tolerance) {
138  // Near |phi|==0, use first order Taylor expansion
139  Rot = Eigen::Matrix3d::Identity() + geometry::skewMatrix(v);
140  } else {
141  Eigen::Vector3d axis = v / angle;
142  Eigen::Matrix3d skew = geometry::skewMatrix(axis);
143  Rot = Eigen::Matrix3d::Identity() + (1. - cos(angle)) * skew * skew +
144  sin(angle) * skew;
145  }
146  return Rot;
147 }

149 inline Eigen::Vector3d log_so3(const Eigen::Matrix3d &M) {
150  double tolerance = 1e-9;
151  double cos_angle = 0.5 * M.trace() - 0.5;
152 
153  // Clip cos(angle) to its proper domain to avoid NaNs from rounding errors
154  cos_angle = std::min(std::max(cos_angle, -1.), 1.);
155  double angle = acos(cos_angle);
156 
157  Eigen::Vector3d phi;
158  // If angle is close to zero, use first-order Taylor expansion
159  if (std::fabs(sin(angle)) < tolerance || angle < tolerance)
160  phi = 0.5 * geometry::FromskewMatrix(M - M.transpose());
161  else {
162  // Otherwise take the matrix logarithm and return the rotation vector
163  phi = (0.5 * angle / sin(angle)) * geometry::FromskewMatrix(M - M.transpose());
164  }
165  return phi;
166 }

242 
243  // bilinear interpolation
244  double dx = pt.x() - (int)pt.x();
245  double dy = pt.y() - (int)pt.y();
246 
247  double weight_tl = (1.0 - dx) * (1.0 - dy);
248  double weight_tr = (dx) * (1.0 - dy);
249  double weight_bl = (1.0 - dx) * (dy);

255  weight_tr * img.at<uint>((int)pt.x() + 1, (int)pt.y()) +
256  weight_bl * img.at<uint>((int)pt.x(), (int)pt.y() + 1) +
257  weight_br * img.at<uint>((int)pt.x() + 1, (int)pt.y() + 1);
258

270 // Check if a point is in a triangle with the barycentric technique
271 // cf. https://blackpawn.com/texts/pointinpoly/
272 template <typename Derived = Eigen::VectorXd> inline bool pointInTriangle(Derived pt, std::vector<Derived> triangle) {
273  // Compute vectors
274  Derived v0 = triangle.at(2) - triangle.at(0);
275  Derived v1 = triangle.at(1) - triangle.at(0);
276  Derived v2 = pt - triangle.at(0);
277 
278  // Compute dot products
279  double dot00 = v0.dot(v0);
280  double dot01 = v0.dot(v1);
281  double dot02 = v0.dot(v2);
282  double dot11 = v1.dot(v1);
283  double dot12 = v1.dot(v2);
284 
285  // Compute barycentric coordinates
286  double inv_denom = 1 / (dot00 * dot11 - dot01 * dot01);
287  double u = (dot11 * dot02 - dot01 * dot12) * inv_denom;
288  double v = (dot00 * dot12 - dot01 * dot02) * inv_denom;
289 
290  // Check if point is in triangle
291  return (u >= 0) && (v >= 0) && (u + v < 1);
292 }

296 inline Eigen::Matrix2d cov2dTriangle(std::vector<Eigen::Vector2d> triangle) {
297 
298  // Compute transformation
299  Eigen::Vector2d x0 = triangle.at(0);
300  Eigen::Matrix2d At;
301  At << triangle.at(1).x() - x0.x(), triangle.at(2).x() - x0.x(), triangle.at(1).y() - x0.y(),
302  triangle.at(2).y() - x0.y();
303  Eigen::Vector2d barycenter = (1.0 / 3.0) * (x0 + triangle.at(1) + triangle.at(2));
304 
305  // Primitive covariance
306  Eigen::Matrix2d Mot;
307  Mot << 1.0 / 12, 1.0 / 24, 1.0 / 24, 1.0 / 12;
308  Eigen::Matrix2d M = 2 * At * Mot * At.transpose() +
309  (barycenter * x0.transpose() + x0 * barycenter.transpose() - x0 * x0.transpose()) -
310  barycenter * barycenter.transpose();
311 
312  return M;
313 }
